;1 _start
section .data
    msg: db 'HELLO, World', 10
    len equ $ - msg

section .text
    global _start










input:
    push rbp
    mov  rbp, rsp
    input_str buffer_for_input, buffer_size
    lea  rdi, [buffer_for_input]
    mov  r11, 0
    mov  al, [rdi]
    cmp  al, '-'
    jne  .check_double
    mov  r11, 1
    inc  rdi

.check_double:
    call check_if_double
    cmp  rax, 0
    je   .int_number
    jmp  .double_number

_start:
        mov rax, 1
    mov rdi, 1
    mov rsi, msg
    mov rdx, len
    syscall
        mov rax, 1
    mov rdi, 1
    mov rsi, msg
    mov rdx, len
    syscall

    mov rax, 60
    xor rdi, rdi
